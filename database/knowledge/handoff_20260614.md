## DriveからPDFをダウンロードする手順（セッション内で再現する方法）

MCPの `download_file_content` は60MB超のPDFをコンテキスト経由で渡せないため、
以下のPythonコードでAnthropicプロキシに直接HTTP接続してダウンロードする。

```python
# Step 1: SSEレスポンスをファイルに保存
import requests

SESSION_TOKEN = open('/home/claude/.claude/remote/.session_ingress_token').read().strip()
SESSION_UUID = "cse_01UUPTckteGxkyNrEU3LrzqY"  # ← 毎セッション変わる
MCP_SERVER_ID = "2dd72e1c-9bcf-46d6-b709-8d38604a8050"

# SESSION_UUIDは /tmp/mcp-config-cse_*.json のファイル名から取得する
# MCP_SERVER_IDは同ファイルのGoogle_DriveサーバーのX-MCP-Server-IDヘッダー

MCP_URL = f"https://api.anthropic.com/v2/ccr-sessions/{SESSION_UUID}/mcp?mcp_url=https%3A%2F%2Fdrivemcp.googleapis.com%2Fmcp%2Fv1&mcp_server_id=1c9babca-934b-5827-b5cf-56e1a89b2e0b&toolbox_mcp_server_id={MCP_SERVER_ID}"

headers = {
    "Authorization": f"Bearer {SESSION_TOKEN}",
    "Content-Type": "application/json",
    "X-Session-UUID": SESSION_UUID,
    "X-MCP-Server-ID": MCP_SERVER_ID,
    "anthropic-version": "2023-06-01",
}
payload = {
    "jsonrpc": "2.0", "id": 2, "method": "tools/call",
    "params": {"name": "download_file_content", "arguments": {"fileId": "<FILE_ID>"}}
}
with requests.post(MCP_URL, headers=headers, json=payload, timeout=300, stream=True) as r:
    with open("/tmp/sse_response.tmp", "wb") as f:
        for chunk in r.iter_content(chunk_size=65536):
            f.write(chunk)

# Step 2: base64抽出してPDF書き込み
import re, base64
content = open("/tmp/sse_response.tmp", encoding="utf-8", errors="replace").read()
m = re.search(r'"content"\s*:\s*"(JVBERi[A-Za-z0-9+/=\\n]+)"', content)
b64 = m.group(1).replace('\\n', '')
open("/tmp/output.pdf", "wb").write(base64.b64decode(b64))
```

### セッションごとに変わる値の取得方法
```bash
# SESSION_UUID
ls /tmp/mcp-config-cse_*.json
# → /tmp/mcp-config-cse_XXXXXXXX.json の XXXXXXXX 部分

# MCP_SERVER_ID (Google_Drive の toolbox_mcp_server_id)
cat /tmp/mcp-config-cse_*.json | python3 -c "
import sys,json; d=json.load(sys.stdin)
cfg = d['mcpServers']['Google_Drive']
# URLからtoolbox_mcp_server_idを抽出
import urllib.parse
url = cfg['url']
qs = urllib.parse.parse_qs(urllib.parse.urlparse(url).query)
print('toolbox_mcp_server_id:', qs.get('toolbox_mcp_server_id', [''])[0])
"
```
